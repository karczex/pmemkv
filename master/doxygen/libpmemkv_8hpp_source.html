275 /*
276  * Abstracts unique_ptr - exposes only void *get() method and a destructor.
277  * This class is needed for C callbacks which cannot be templated
278  * (type of object and deleter must be abstracted away).
279  */
280 struct unique_ptr_wrapper_base {
281  virtual ~unique_ptr_wrapper_base()
282  {
283  }
284 
285  virtual void *get() = 0;
286 };
287 
288 template <typename T, typename D>
289 struct unique_ptr_wrapper : public unique_ptr_wrapper_base {
290  unique_ptr_wrapper(std::unique_ptr<T, D> ptr) : ptr(std::move(ptr))
291  {
292  }
293 
294  void *get() override
295  {
296  return ptr.get();
297  }
298 
299  std::unique_ptr<T, D> ptr;
300 };

530 template <typename Comparator>
531 inline status config::put_comparator(Comparator &&comparator)
532 {
533  static_assert(
534  std::is_same<decltype(std::declval<Comparator>().compare(
535  std::declval<string_view>(),
536  std::declval<string_view>())),
537  int>::value,
538  "Comparator should implement `int compare(pmem::kv::string_view, pmem::kv::string_view)` method");
539  static_assert(std::is_convertible<decltype(std::declval<Comparator>().name()),
540  std::string>::value,
541  "Comparator should implement `std::string name()` method");
542 
543  std::unique_ptr<internal::comparator_base> wrapper;
544 
545  try {
546  wrapper = std::unique_ptr<internal::comparator_base>(
547  new internal::comparator_wrapper<Comparator>(
548  std::forward<Comparator>(comparator)));
549  } catch (std::bad_alloc &e) {
550  return status::OUT_OF_MEMORY;
551  } catch (...) {
552  return status::UNKNOWN_ERROR;
553  }
554 
555  auto cmp =
556  std::unique_ptr<pmemkv_comparator, decltype(pmemkv_comparator_delete) *>(
557  pmemkv_comparator_new(&internal::call_comparator_function,
558  std::string(comparator.name()).c_str(),
559  wrapper.get()),
560  &pmemkv_comparator_delete);
561  if (cmp == nullptr)
562  return status::UNKNOWN_ERROR;
563 
564  internal::unique_ptr_wrapper_base *entry;
565 
566  try {
567  entry = new internal::comparator_config_entry(std::move(wrapper),
568  std::move(cmp));
569  } catch (std::bad_alloc &e) {
570  return status::OUT_OF_MEMORY;
571  } catch (...) {
572  return status::UNKNOWN_ERROR;
573  }
574 
575  return static_cast<status>(pmemkv_config_put_object_cb(
576  this->_config, "comparator", (void *)entry, internal::call_up_get,
577  internal::call_up_destructor));
578 }
